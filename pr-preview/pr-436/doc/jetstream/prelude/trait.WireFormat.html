pub trait WireFormat: Send {
    // Required methods
    fn byte_size(&self) -> u32;

       where W: Write,
             Self: Sized;
    fn decode<R>(reader: &mut R) -> Result<Self, Error>
       where R: Read,
             Self: Sized;
}
Expand description

A type that can be encoded on the wire using the 9P protocol.
