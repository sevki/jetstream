pub trait WireFormat: Send {
    // Required methods
    fn byte_size(&self) -> u32;
    fn encode<W: Write>(&self, writer: &mut W) -> Result<()>
       where Self: Sized;

       where Self: Sized;
}
Expand description

A type that can be encoded on the wire using the 9P protocol.
